class Complex_No {
    float real, img;
 //creation of a class inside it defining 2 parts of the object complex no that is real
    // and imaginary part
   public Complex_No() {
//intializing 2 float real with zero after that we can store valur in them;

        real = 0;
        img = 0;
    }

    public Complex_No(float real , float img) {
// passing function with 2 nos in it
        this.real =real;
        this.img = img;
    }

    public void Display(Complex_No C1, Complex_No C2) {
        //created a display function in it which will print my two complex nos

    public void AddNumbers(Complex_No C1, Complex_No C2) {
        // created a add function to add two diff nos in it with creating a float
        // and then cout them in my main function
        float real, img;
        real = (C1.real + C2.real);
        img = (C1.img + C2.img);
        System.out.println("Addition of Complex Numbers =(" + real + ")+(" + img + ")i");
    }

    public void SubNumbers(Complex_No C1, Complex_No C2) {
        // created a Subtraction function to subtract two diff nos in it with creating a float
        // and then cout them in my main function
        float real, img;
        real = (C1.real - C2.real);
        img = (C1.img - C2.img);
        System.out.println("Substraction of Complex Numbers =(" + real + ")+(" + img + ")i");
    }

    public void MultiNumbers(Complex_No C1, Complex_No C2) {
        // created a Multiplicative  function to subtract two diff nos in it with creating a float
        // and then miultiply them in my main function
        float real, img;
        real = (C1.real * C2.real - C1.img * C2.img);
        img = (C1.real * C2.img + C1.img * C2.real);
        System.out.println("Multiplication of Complex Numbers =(" + real + ")+(" + img + ")i");
    }

    public void DivNumbers(Complex_No C1, Complex_No C2) {
        float real, img;

       // first i created a if condition if my second complex no is 0 then it will print not divisible
        if (C2.real == 0 && C2.img==0){
            System.out.println("Division not possible");
        }
        else {
        real = (C1.real * C2.real + C1.img * C2.img) / (C2.real * C2.real + C2.img * C2.img);
        img = (C1.img * C2.real - C1.real * C2.img) / (C2.real * C2.real + C2.img * C2.img);
        System.out.println("Division of Complex Numbers =(" + real + ")+(" + img + ")i");
    }}
}
