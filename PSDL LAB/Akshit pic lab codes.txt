#include<stdio.h>
#include<stdlib.h>
#include<pic18f4550.h>

 int div,mult,sum,sub;
 div= 0x0C / 0x04;
 TRISC=0;
 PORTC=div;
 sum = 0x0B + 0x04;
 TRISA=0;
 PORTA = sum;
 mult= 0x0A * 0x02;
 TRISD=0;

 sub= 0x0A - 0x03;

Assignment by Akshit 3 :--- Ascending and descending 


#include<xc.h>
#include<stdio.h>
#include<stdlib.h>
#include<pic18f4550.h>
#include<stdint.h>
void main (void){
 int arr[7]={0x02,0x0A,0x05,0x0E,0x04,0x0C,0x08};
 for(int i=0;i<6;i++){
 for(int j=i+1;j<7;j++){
 if(arr[j]<arr[i]){
 int temp;
 temp=arr[i];
 arr[i]=arr[j];
 arr[j]=temp;
 }
 }
 }
}

Assignment by Akshit 6 :--  buzzer 



#include <xc.h>
#include <pic18f4550.h>
#define Buzzer LATAbits.LATA5 
unsigned int count = 0;
void __interrupt(high_priority) Timer1_ISR(void)
{
    if (TMR1IF == 1)
    {
        TMR1L = 0x20;
        TMR1H = 0xD1;
          count++;
        if (count >= 1000) 
        {
            Buzzer = ~Buzzer; 
            count = 0;       
        }

        TMR1IF = 0;
    }
}

void main()
{
    TRISB = 0;             
    TRISAbits.TRISA5 = 0;  
    GIE = 1;     
    PEIE = 1;   
    TMR1IE = 1;  
    TMR1IF = 0;  
    T1CON = 0b00000010; 
    TMR1L = 0x20; 
    TMR1H = 0xD1;
    TMR1ON = 1; 

    while (1);
}

#include <xc.h>

#define RELAY_PIN LATAbits.LATA4

void __interrupt(high_priority) extint_isr(void)
{
    unsigned int i;
    if (INT1F)
    {
        INT1F = 0; 
        INT1IE = 0;
        RELAY_PIN = ~RELAY_PIN; 
        // Debouncing delay
        for (i = 0; i < 10000; i++);

        INT1IE = 1;     }
}

{
    ADCON1 = 0x0F;
