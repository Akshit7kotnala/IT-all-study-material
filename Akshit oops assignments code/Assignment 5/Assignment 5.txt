import java.util.Scanner;

interface vehicle{
    void speedup();
    void apply_break();
    void change_gear(int x);
    void display();
}

 class car implements vehicle{
    int current_speed;
    int current_gear;
    final int MAX_GEAR;
     final int MAX_SPEED;
    car(){
          MAX_SPEED = 150;
          MAX_GEAR = 5;
         current_speed = 0;
         current_gear = 0;
    }

     public void speedup(){
        current_speed += 30;
        current_gear += 1;
        System.out.println("Current speed of the vehicle is : "+current_speed);
        if(current_speed>=MAX_SPEED ) {
            current_gear=MAX_GEAR;
            current_speed=MAX_SPEED;
            System.out.println("MAX speed limit reached!!!");
        }

    }
    public void apply_break() {
       current_speed=0;
       current_gear=0;
    }
    public void change_gear(int x){
        if(x==2){
            current_gear+=1;
            current_speed+=30;
           
        }
        else if(x==3){
            current_speed-=30;
            current_gear-=1;
        }
         System.out.println("Current gear is :"+current_gear);
        if(current_gear>=MAX_GEAR){
            current_gear=MAX_GEAR;
            current_speed=MAX_SPEED;
            System.out.println("Car is at max gear!!!");
        }
    }
    public void display(){
        System.out.println("Current speed of car is "+current_speed);
        System.out.println("Current gear of car is "+current_gear);
    }
}
