#include<bits/stdc++.h>
using namespace std;


bool isEmpty(int r, int f){
	return f==-1 and r==-1;
}
bool isFull(int r, int f, int n ){
	return (r+1)%n==f;
}





void enqueue(int a[],int&r , int &f , int n , int v){
	if (isFull(r,f,n)){
		cout<<"Queue is full"<<endl;
		return;
	}
	if(isEmpty(r,f)){
		f=r=0;
	}
	else {r=(r+1)%n;}
	a[r]=v;
	cout<<"Enqued "<<v<<endl;
	
}

void dequeue(int a[],int&r, int &f, int n ){
	if(isEmpty(r,f)){
		cout<<"Queue is Empty"<<endl;
		return;
	}
	int deqval=a[f];
	if(f==r){
		f=r=-1;
	}
	else {f=(f+1)%n;}
	cout<<"Dequqed "<<deqval<<endl;
}

void display (int a[], int r , int f , int n ){
	if(isEmpty(r,f)){
		cout<<"Queue is empty "<<endl;
		return;
	}
	cout<<"Queue elements are : ";
	int i = f ; 
	do {
		cout<<a[i]<<" ";
		i=(i+1)%n;
	}while (i!=(r+1)%n);
	cout<<endl;
}
